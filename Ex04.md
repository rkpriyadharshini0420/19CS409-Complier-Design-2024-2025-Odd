**parser.y**
```
%{
#include <stdio.h>
#include <stdlib.h>

extern int yylex();
extern int yyparse();
extern char* yytext;

%}

%token VAR NUM

%%

program:
    | program statement
    ;

statement:
    VAR {
        printf("Valid variable: %s\n", yytext);
    }
    | NUM {
        printf("Valid number: %s\n", yytext);
    }
    ;

%%

int main(void) {
    printf("Enter a variable or number:\n");
    yyparse();
    return 0;
}

int yyerror(const char *msg) {
    fprintf(stderr, "Error: %s\n", msg);
    return 0;
}


```
